rules_version = '2';

service cloud.firestore {
  match /databases/{database}/documents {

    function signedIn() {
      return request.auth != null;
    }

    function isAdmin() {
      return signedIn() && (
        exists(/databases/$(database)/documents/admins/$(request.auth.uid)) ||
        request.auth.token.email == 'rabbihossainltd@gmail.com'
      );
    }

    function isOwner(userId) {
      return signedIn() && request.auth.uid == userId;
    }

    function positiveNumber(value) {
      return (value is int || value is float) && value > 0;
    }

    function hasAmountUsd() {
      return (
        request.resource.data.keys().hasAny(['amountUsd']) &&
        positiveNumber(request.resource.data.amountUsd)
      ) || (
        request.resource.data.keys().hasAny(['amountUSD']) &&
        positiveNumber(request.resource.data.amountUSD)
      );
    }

    function hasAmountBdt() {
      return (
        request.resource.data.keys().hasAny(['amountBdt']) &&
        positiveNumber(request.resource.data.amountBdt)
      ) || (
        request.resource.data.keys().hasAny(['amountBDT']) &&
        positiveNumber(request.resource.data.amountBDT)
      );
    }

    function hasValidRate() {
      return (
        request.resource.data.keys().hasAny(['rate']) &&
        request.resource.data.rate == 125
      ) || (
        request.resource.data.keys().hasAny(['rateBDT']) &&
        request.resource.data.rateBDT == 125
      );
    }

    // User profile create: credit fields must start at 0
    function validCreditCreate() {
      return
        (!request.resource.data.keys().hasAny(['credit']) || request.resource.data.credit == 0) &&
        (!request.resource.data.keys().hasAny(['creditUSD']) || request.resource.data.creditUSD == 0) &&
        (!request.resource.data.keys().hasAny(['creditBDT']) || request.resource.data.creditBDT == 0);
    }

    // User can only decrease their own credit (deduction on purchase)
    function creditIsSafe() {
      return !request.resource.data.diff(resource.data).affectedKeys().hasAny(['credit']) ||
        (
          resource.data.keys().hasAny(['credit']) &&
          (resource.data.credit is int || resource.data.credit is float) &&
          (request.resource.data.credit is int || request.resource.data.credit is float) &&
          request.resource.data.credit <= resource.data.credit &&
          request.resource.data.credit >= 0
        ) ||
        (
          !resource.data.keys().hasAny(['credit']) &&
          request.resource.data.credit == 0
        );
    }

    function creditUSDIsSafe() {
      return !request.resource.data.diff(resource.data).affectedKeys().hasAny(['creditUSD']) ||
        (
          resource.data.keys().hasAny(['creditUSD']) &&
          (resource.data.creditUSD is int || resource.data.creditUSD is float) &&
          (request.resource.data.creditUSD is int || request.resource.data.creditUSD is float) &&
          request.resource.data.creditUSD <= resource.data.creditUSD &&
          request.resource.data.creditUSD >= 0
        ) ||
        (
          !resource.data.keys().hasAny(['creditUSD']) &&
          request.resource.data.creditUSD == 0
        );
    }

    function creditBDTIsSafe() {
      return !request.resource.data.diff(resource.data).affectedKeys().hasAny(['creditBDT']) ||
        (
          resource.data.keys().hasAny(['creditBDT']) &&
          (resource.data.creditBDT is int || resource.data.creditBDT is float) &&
          (request.resource.data.creditBDT is int || request.resource.data.creditBDT is float) &&
          request.resource.data.creditBDT <= resource.data.creditBDT &&
          request.resource.data.creditBDT >= 0
        ) ||
        (
          !resource.data.keys().hasAny(['creditBDT']) &&
          request.resource.data.creditBDT == 0
        );
    }

    // Only status+timestamp fields can change on cancel
    function onlyCancelFields() {
      return request.resource.data.diff(resource.data).affectedKeys()
        .hasOnly(['status', 'cancelledAt', 'updatedAt']);
    }

    // Service order create: user creates own order with credit or admin creates instant_pay order
    function validServiceOrderCreate() {
      return signedIn()
        && request.resource.data.userId is string
        && request.resource.data.serviceName is string
        && hasAmountUsd()
        && request.resource.data.paymentMethod in ['credit', 'instant_pay']
        && (isAdmin() || request.resource.data.userId == request.auth.uid);
    }

    // ─── admins collection ───────────────────────────────────────────────────
    match /admins/{adminId} {
      allow read: if signedIn() && (request.auth.uid == adminId || isAdmin());
      allow write: if false; // only via Firebase Console
    }

    // ─── users/{userId} ──────────────────────────────────────────────────────
    // Admin: full read/write (can increase or decrease credit)
    // Owner: read own profile; update allowed only if credit is not increased
    match /users/{userId} {
      allow read: if isOwner(userId) || isAdmin();
      allow create: if isOwner(userId) && validCreditCreate();
      allow update: if isAdmin() || (
        isOwner(userId) &&
        creditIsSafe() &&
        creditUSDIsSafe() &&
        creditBDTIsSafe()
      );
      allow delete: if false;
    }

    // ─── topups/{topupId} ────────────────────────────────────────────────────
    // User creates own pending topup (credit top-up or instant service payment)
    // Admin can approve/decline
    // User can cancel own pending topup
    match /topups/{topupId} {
      allow create: if signedIn()
        && request.resource.data.userId == request.auth.uid
        && request.resource.data.status == 'pending'
        && hasAmountUsd()
        && hasAmountBdt()
        && hasValidRate()
        && request.resource.data.method in ['bKash', 'Nagad', 'Rocket']
        && request.resource.data.transactionId is string
        && request.resource.data.transactionId.size() >= 5;

      allow read: if signedIn() && (resource.data.userId == request.auth.uid || isAdmin());

      allow update: if isAdmin() || (
        signedIn() &&
        resource.data.userId == request.auth.uid &&
        resource.data.status == 'pending' &&
        request.resource.data.status == 'cancelled' &&
        onlyCancelFields()
      );

      allow delete: if false;
    }

    // ─── serviceOrders/{orderId} ─────────────────────────────────────────────
    // Created by user (credit purchase) or by admin (instant pay confirm)
    // Admin can accept/decline
    // User can cancel own pending order
    match /serviceOrders/{orderId} {
      allow create: if validServiceOrderCreate();
      allow read: if signedIn() && (resource.data.userId == request.auth.uid || isAdmin());

      allow update: if isAdmin() || (
        signedIn() &&
        resource.data.userId == request.auth.uid &&
        resource.data.status in ['pending', 'paid_credit_pending_review', 'paid_instant_pending_review'] &&
        request.resource.data.status == 'cancelled' &&
        onlyCancelFields()
      );

      allow delete: if false;
    }

    // ─── orders/{orderId} (legacy compat) ────────────────────────────────────
    match /orders/{orderId} {
      allow create: if validServiceOrderCreate();
      allow read: if signedIn() && (resource.data.userId == request.auth.uid || isAdmin());

      allow update: if isAdmin() || (
        signedIn() &&
        resource.data.userId == request.auth.uid &&
        resource.data.status in ['pending', 'paid_credit_pending_review', 'paid_instant_pending_review'] &&
        request.resource.data.status == 'cancelled' &&
        onlyCancelFields()
      );

      allow delete: if false;
    }
  }
}

//   Users collection:
//     credit        (number, USD) — primary balance field
//     creditUSD     (number, USD) — duplicate for legacy compat
//     creditBDT     (number, BDT = credit * 125) — BDT display
//
//   serviceOrders collection:
//     amountUsd     (number)
//     amountUSD     (number) — duplicate for legacy compat
//     amountBdt     (number)
//     amountBDT     (number) — duplicate
//     rate          (125)
//     paymentMethod ("credit" | "instant_pay")
//     status        ("pending" | "paid_credit_pending_review" | "paid_instant_pending_review"
//                    | "accepted" | "declined" | "cancelled")
//
//   topups collection:
//     amountUsd / amountUSD / amountBdt / amountBDT
//     rate / rateBDT (125)
//     method ("bKash" | "Nagad" | "Rocket")
//     status ("pending" | "approved" | "declined" | "cancelled")
//     purpose ("credit" | "service")
//
// USD → BDT rate: 1 USD = 125 BDT
// Display format: $1 / ৳125
